selectionSort(arr, n)
begin
    for i ← 0 to n-2 do
    begin
        minIdx ← i
        for j ← i+1 to n-1 do
        begin
            if (arr[j] < arr[minIdx]) then
            begin
                minIdx ← j
            end
        end
        temp ← arr[i]
        arr[i] ← arr[minIdx]
        arr[minIdx] ← temp
    end
end
